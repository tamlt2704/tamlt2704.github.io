def raw_data_to_df(k, value):
    mo = re.findall('(.*?)[\s]*(\([\d,]*\))', value)
    data = [[] for _ in range(2)]
    for title, count in mo:
        title = title.strip()
        count = int(re.sub("\D", "", count))
        data[0].append(title)
        data[1].append(count)
    df = pd.DataFrame(data).T
    df.columns = [k, '{}_count'.format(k)]    
    return df
